  9class ImageEncoder(nn.Module):
 10    """
 11    ImageEncoder class using a pre-trained ResNet18 model for feature extraction,
 12    with built-in image preprocessing.
 13
 14    Methods:
 15    --------
 16    preprocess_image(image: torch.Tensor) -> torch.Tensor
 17        Applies preprocessing to an input image.
 18    forward(image: torch.Tensor) -> torch.Tensor
 19        Passes the preprocessed input image through the CNN to extract feature maps.
 20    """
 21
 22    def __init__(self):
 23        super(ImageEncoder, self).__init__()
 24        # uses a pre-trained ResNet18 model, excluding the last fully connected layer
 25        self.cnn = models.resnet18(weights=ResNet18_Weights.IMAGENET1K_V1)
 26        self.cnn = nn.Sequential(*list(self.cnn.children())[:-2])
 27
 28        self.preprocess = transforms.Compose([
 29            transforms.Resize((224, 224)),
 30            transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225])  # normalize with ImageNet stats
 31        ])
 32
 33    def preprocess_image(self, image: torch.Tensor) -> torch.Tensor:
 34        """
 35        Preprocesses the input image by resizing and normalizing.
 36
 37        Parameters:
 38        -----------
 39        image : torch.Tensor
 40            Input image tensor of shape (C, H, W).
 41
 42        Returns:
 43        --------
 44        torch.Tensor
 45            Preprocessed image tensor of shape (1, 3, 224, 224), with a batch dimension.
 46        """
 47        preprocessed_image = self.preprocess(image)
 48        return preprocessed_image
 49
 50    def forward(self, image: torch.Tensor) -> torch.Tensor:
 51        """
 52        Forward pass to extract features from the input image.
 53
 54        Parameters:
 55        -----------
 56        image : torch.Tensor
 57            Input image in tensor format.
 58
 59        Returns:
 60        --------
 61        torch.Tensor
 62            Output tensor of shape (batch_size, 512, H, W), where 512 is the number of feature
 63            channels, and H and W are the spatial dimensions of the output feature map.
 64        """
 65        preprocessed_image = self.preprocess_image(image)
 66        features = self.cnn(preprocessed_image)
 67        return features

 69class QuestionEncoder(nn.Module):
 70    """
 71    Module for encoding questions using an LSTM.
 72
 73
 74    Methods:
 75    -------
 76    forward(questions):
 77        Performs a forward pass through the network, returning the final hidden state of the LSTM.
 78    """
 79
 80    def __init__(self, vocab_size, embed_size, hidden_size, num_layers):
 81        super(QuestionEncoder, self).__init__()
 82        self.embedding = nn.Embedding(vocab_size, embed_size)
 83        self.lstm = nn.LSTM(embed_size, hidden_size, num_layers, batch_first=True)
 84
 85    def forward(self, questions):
 86        """
 87        Performs a forward pass through the network.
 88
 89        Parameters
 90        ----------
 91        questions : torch.Tensor
 92            A batch of tokenized questions represented as tensors.
 93
 94        Returns
 95        -------
 96        torch.Tensor
 97            The last hidden state of the LSTM for each question in the batch, with shape (batch_size, hidden_size).
 98        """
 99        
100        embedded = self.embedding(questions)  # shape: (batch_size, seq_len, embed_size)
101        _, (hidden, _) = self.lstm(embedded)  # only take the hidden state output
102
103        # return the last hidden state of the LSTM
104        return hidden[-1].unsqueeze(0)  # shape: (batch_size, hidden_size)

106class RelationalNetwork(nn.Module):
107    """
108    Module for relational reasoning, composed of two MLPs (g_theta and f_phi).
109
110    Attributes
111    ----------
112    g_theta : nn.Sequential
113        An MLP that takes concatenated object features and question embeddings to learn relations.
114    f_phi : nn.Sequential
115        A final MLP that takes the summed output of all g_theta outputs to produce a final prediction.
116
117    Methods
118    -------
119    forward(object_features, question_embedding):
120        Performs a forward pass through the network using object features and a question embeddings.
121    """
122
123    def __init__(self, feature_dim, question_dim, g_theta_dim, f_phi_dim, num_classes):
124        """
125        Initializes the RelationalNetwork with two MLPs: g_theta and f_phi.
126
127        Parameters
128        ----------
129        feature_dim : int
130            The dimensionality of the object features.
131        question_dim : int
132            The dimensionality of the question embedding.
133        g_theta_dim : int
134            The dimensionality of the hidden layer in the g_theta MLP.
135        f_phi_dim : int
136            The dimensionality of the hidden layer in the f_phi MLP.
137        num_classes : int
138            The number of output classes for the multi-class classification.        
139        """
140        super(RelationalNetwork, self).__init__()
141        
142        self.g_theta = nn.Sequential(
143            nn.Linear(2 * feature_dim + question_dim, g_theta_dim),
144            nn.ReLU(),
145            nn.Linear(g_theta_dim, g_theta_dim),
146            nn.ReLU()
147        )
148        self.f_phi = nn.Sequential(
149            nn.Linear(g_theta_dim, f_phi_dim),
150            nn.ReLU(),
151            nn.Linear(f_phi_dim, num_classes)
152        )
153
154    def forward(self, object_features, question_embedding):
155        """
156        Performs a forward pass through the network.
157
158        Parameters
159        ----------
160        object_features : torch.Tensor
161            A tensor containing the features of objects, with shape (batch_size, num_objects, feature_dim).
162        question_embedding : torch.Tensor
163            A tensor representing the question embedding, with shape (batch_size, question_dim).
164
165        Returns
166        -------
167        torch.Tensor
168            The output tensor, which contains the prediction result. The shape is (batch_size, num_classes).
169        """
170        num_objects = object_features.size(1)
171        relations = []
172        for i in range(num_objects):
173            for j in range(num_objects):
174                if i != j:
175                    # concatenate the features of object pairs with the question embedding
176                    pair_features = torch.cat(
177                        [object_features[:, i], object_features[:, j], question_embedding], dim=1
178                    )
179                    relations.append(self.g_theta(pair_features))
180        
181        # sum the outputs from g_theta
182        relations_sum = torch.stack(relations, dim=1).sum(dim=1)
183        # apply f_phi to the summed output
184        output = self.f_phi(relations_sum)
185        
186        return output

188class RelationalReasoningModel(nn.Module):
189    """
190    Model for visual question answering using relational reasoning.
191
192    This model consists of three main components:
193    - ImageEncoder: encodes the input image into object features
194    - QuestionEncoder: encodes the input question into a vector embedding
195    - RelationalNetwork: performs relational reasoning over object features and the question embedding
196
197    Methods
198    -------
199    forward(image, question):
200        Performs a forward pass through the network, processing the image and question to produce an output.
201    """
202
203    def __init__(self, vocab_size, embed_size, hidden_size, num_layers, feature_dim, g_theta_dim, f_phi_dim, num_classes):
204        """
205        Initializes the RelationalReasoningModel with an ImageEncoder, QuestionEncoder, and RelationalNetwork.
206
207        Parameters
208        ----------
209        vocab_size : int
210            The size of the vocabulary used in the QuestionEncoder.
211        embed_size : int
212            The dimensionality of the word embeddings in the QuestionEncoder.
213        hidden_size : int
214            The number of features in the hidden state of the LSTM in the QuestionEncoder.
215        num_layers : int
216            The number of recurrent layers in the LSTM of the QuestionEncoder.
217        feature_dim : int
218            The dimensionality of the object features produced by the ImageEncoder.
219        g_theta_dim : int
220            The dimensionality of the hidden layers in the g_theta MLP of the RelationalNetwork.
221        f_phi_dim : int
222            The dimensionality of the hidden layers in the f_phi MLP of the RelationalNetwork.
223        num_classes : int
224            The number of output classes for the multi-class classification.
225        """
226        super(RelationalReasoningModel, self).__init__()
227        self.image_encoder = ImageEncoder()
228        self.question_encoder = QuestionEncoder(vocab_size, embed_size, hidden_size, num_layers)
229        self.relation_network = RelationalNetwork(feature_dim, hidden_size, g_theta_dim, f_phi_dim, num_classes)
230    
231    def forward(self, image, question):
232        """
233        Performs a forward pass through the model.
234
235        Parameters
236        ----------
237        image : torch.Tensor
238            A tensor representing the input image of shape (batch_size, channels, height, width).
239        question : torch.Tensor
240            A tensor representing the input question of shape (batch_size, seq_len).
241
242        Returns
243        -------
244        torch.Tensor
245            The output tensor, representing the predicted class logits, with shape (batch_size, num_classes).
246        """
247        image_features = self.image_encoder(image)
248        # flatten the spatial dimensions to produce a set of object features
249        object_features = image_features.view(image_features.size(0), -1, image_features.size(1))
250        question_embedding = self.question_encoder(question)
251        # perform relational reasoning using the object features and question embedding
252        output = self.relation_network(object_features, question_embedding)
253        
254        return output

256class BaselineModel(nn.Module):
257    """
258    Base model that combines a CNN-based image encoder and an LSTM-based question encoder
259    with a simple MLP for classification without a relational network module.
260    """
261
262    def __init__(self, vocab_size, embed_size, hidden_size, num_layers, num_classes):
263        super(BaselineModel, self).__init__()
264        self.image_encoder = ImageEncoder()
265        self.question_encoder = QuestionEncoder(vocab_size, embed_size, hidden_size, num_layers)
266        
267        # MLP for classification
268        self.fc1 = nn.Linear(512 * 7 * 7 + hidden_size, 256)  # where 512 is the number of channels in the output feature map of ResNet18 and 7x7 are the spatial dimensions, hidden_size comes from the dimensionality of the question embeddings
269        self.fc2 = nn.Linear(256, num_classes)
270
271    def forward(self, image: torch.Tensor, questions: torch.Tensor) -> torch.Tensor:
272        """
273        Forward pass to get the model's prediction.
274        
275        Parameters:
276        -----------
277        image : torch.Tensor
278            Input image in tensor format.
279        questions : torch.Tensor
280            Input tensor containing word indices of shape (batch_size, seq_len).
281
282        Returns:
283        --------
284        torch.Tensor
285            Output prediction tensor of shape (batch_size, num_classes).
286        """
287        
288        # encode the image and flatten the features
289        image_features = self.image_encoder(image)  # shape: (batch_size, 512, 7, 7)
290        
291        image_features = image_features.view(image_features.size(0), -1)  # flatten to (batch_size, 512*7*7)
292        
293        # encode the question
294        question_embedding = self.question_encoder(questions)  # shape: (batch_size, hidden_size)
295        
296        # concatenate image and question features
297        combined_features = torch.cat((image_features, question_embedding), dim=1)  # shape: (batch_size, 512*7*7 + hidden_size)
298        
299        # pass through the MLP
300        x = nn.ReLU()(self.fc1(combined_features))
301        output = self.fc2(x)
302        
303        return output

308class ModelConstructor:
309    """
310    A class to manage and load different types of models: predefined models (BaselineModel, RelationalReasoningModel) 
311    or custom models in ONNX or PyTorch format, please mind the ONNX can be used only for inference.
312
313    Methods:
314    --------
315    load_model(model_type: str, **kwargs) -> nn.Module:
316        Loads a predefined model ('baseline', 'relational') or a custom ONNX/PyTorch model based on the input parameters.
317    """
318
319    def __init__(self):
320        pass
321
322    def load_model(self, model_type: str, **kwargs) -> torch.nn.Module:
323        """
324        Load and initialize one of the predefined models or a custom model.
325
326        Parameters:
327        -----------
328        model_type : str
329            The type of model to load ('baseline', 'relational', 'onnx', or 'custom').
330        **kwargs : dict
331            Additional arguments required to initialize the model (e.g., vocab_size, embed_size, etc.).
332
333        Returns:
334        --------
335        nn.Module or ort.InferenceSession
336            The initialized PyTorch model or ONNX session for inference.
337        """
338        if model_type == 'baseline':
339            return BaselineModel(
340                vocab_size=kwargs['vocab_size'],
341                embed_size=kwargs['embed_size'],
342                hidden_size=kwargs['hidden_size'],
343                num_layers=kwargs['num_layers'],
344                num_classes=kwargs['num_classes']
345            )
346        elif model_type == 'relational':
347            return RelationalReasoningModel(
348                vocab_size=kwargs['vocab_size'],
349                embed_size=kwargs['embed_size'],
350                hidden_size=kwargs['hidden_size'],
351                num_layers=kwargs['num_layers'],
352                feature_dim=kwargs['feature_dim'],
353                g_theta_dim=kwargs['g_theta_dim'],
354                f_phi_dim=kwargs['f_phi_dim'],
355                num_classes=kwargs['num_classes']
356            )
357        elif model_type == 'onnx':
358            return self._load_custom_model_onnx(kwargs['onnx_path'])
359        elif model_type == 'custom':
360            return self._load_custom_model(kwargs['model_class'], kwargs.get('weights_path'))
361        else:
362            raise ValueError(f"Unknown model type: {model_type}")
363
364    def _load_custom_model(self, model_class: torch.nn.Module, weights_path: str = None) -> torch.nn.Module:
365        """
366        @public
367        
368        Loads a custom PyTorch model and optionally loads pre-trained weights.
369
370        Parameters:
371        -----------
372        model_class : torch.nn.Module
373            The class or instance of the custom model to load.
374        weights_path : str, optional
375            Path to the PyTorch model weights file (.pt or .pth) (default=None).
376
377        Returns:
378        --------
379        nn.Module
380            The initialized PyTorch model, optionally with loaded weights.
381        """
382        model = model_class() if isinstance(model_class, type) else model_class
383
384        if weights_path:
385            model.load_state_dict(torch.load(weights_path))
386
387        return model
388
389    def _load_custom_model_onnx(self, onnx_path: str):
390        """
391        @public
392
393        Loads a custom model from an ONNX file using ONNX Runtime.
394
395        Parameters:
396        -----------
397        onnx_path : str
398            Path to the ONNX model file.
399
400        Returns:
401        --------
402        ort.InferenceSession
403            An ONNX Runtime session that can be used to make predictions with the ONNX model.
404        """
405        onnx_model = onnx.load(onnx_path)
406        onnx.checker.check_model(onnx_model)
407        session = ort.InferenceSession(onnx_path)
408
409        return session
