 291        Parameters:

 349        """
 350        @public

 353
 354        Parameters:

 358        max_objects : int
 359            Maximum number of objects to include in the image.

 366        fixed_colors : bool

 370        shapes : dict

 375            Whether colors from the palette can be reused.
 376
 377        Returns:
 378        --------
 379        img : numpy.ndarray
 380            The generated image with the specified number of shapes.
 381        objects_info : list
 382            A list of dictionaries containing information about each shape in the image.
 383        """
 384        img = np.zeros((self.img_dim, self.img_dim, 3))
 385        existing_boxes = []
 386        attempts = 0
 387        max_attempts = max_objects * 10
 388        added_objects = 0
 389        objects_info = []
 390        object_id = 1
 391        colors = list(palette.keys())
 392        
 393        if add_noise:
 394            img = self._gen_noise(img, noise_prob)
 395        
 396        if min_objects == max_objects:
 397            num_objects = min_objects
 398        else:
 399            num_objects = np.random.randint(min_objects, max_objects)
 400        
 401        while added_objects < num_objects:
 402            # choose the color of the object
 403            if fixed_colors:
 404                random_color_key = np.random.choice(colors)
 405                if not reusable_palette_colors:
 406                    colors.remove(random_color_key)

 548    def _find_line_endpoints(self, center_x, center_y, slope, length):
 549        """
 550        @public
 551
 552        Find the endpoints of a line segment given its center, slope, and length.
 553
 554        Parameters:
 555        -----------
 556        center_x : float
 557            The x-coordinate of the center point of the line.
 558        center_y : float

 571        angle = math.atan(slope)
 572

 575        dy = 0.5 * length * math.sin(angle)
 576
 577        # calculate the start and end point coordinates
 578        start_x = int(center_x - dx)
 579        start_y = int(center_y - dy)
 580        end_x = int(center_x + dx)
 581        end_y = int(center_y + dy)

 583        return (start_x, start_y), (end_x, end_y)

 629    def _do_overlap(self, box1, box2):
 630        """
 631        @public
 632
 633        Check if two bounding boxes overlap with an offset applied to all sides.
 634
 635        Parameters:
 636        -----------
 637        box1 : tuple
 638            A tuple containing two points that represent the first bounding box ((x1, y1), (x2, y2)).
 639        box2 : tuple
 640            A tuple containing two points that represent the second bounding box ((x1, y1), (x2, y2)).
 641
 642        Returns:
 643        --------
 644        bool
 645            True if the bounding boxes overlap, False otherwise.
 646        """
 647        # check if two bounding boxes overlap with an offset applied to all sides
 648        offset = 3
 649
 650        expanded_box1 = ((box1[0][0] - offset, box1[0][1] - offset), (box1[1][0] + offset, box1[1][1] + offset))
 651        expanded_box2 = ((box2[0][0] - offset, box2[0][1] - offset), (box2[1][0] + offset, box2[1][1] + offset))
 652
 653        x_overlaps = (expanded_box1[0][0] < expanded_box2[1][0]) and (expanded_box1[1][0] > expanded_box2[0][0])
 654        y_overlaps = (expanded_box1[0][1] < expanded_box2[1][1]) and (expanded_box1[1][1] > expanded_box2[0][1])
 655        
 656        return x_overlaps and y_overlaps

 658    def _add_pairwise_distances(self, objects_info):
 659        """
 660        @public
 661
 662        Add pairwise distances between all objects in the image to their information.
 663
 664        Parameters:
 665        -----------
 666        objects_info : list
 667            A list of dictionaries, where each dictionary contains information about an object 
 668            in the image, including its bounding box location.
 669
 670        Returns:
 671        --------
 672        list
 673            The updated list of dictionaries with pairwise distances added for each object.
 674        """
 675        for obj_info in objects_info:
 676            # skip the question object
 677            if 'question' in obj_info:
 678                continue
 679            bbox_center = self._compute_bbox_center(obj_info["bboxlocation"])
 680            distances = []
 681            for other_obj_info in objects_info:
 682                # skip the question object
 683                if 'question' in other_obj_info:
 684                    continue
 685                if other_obj_info["id"] != obj_info["id"]:  # don't compare with itself
 686                    other_bbox_center = self._compute_bbox_center(other_obj_info["bboxlocation"])
 687                    distance = self._euclidean_distance(bbox_center, other_bbox_center)
 688                    distances.append({
 689                        "object_id": other_obj_info["id"],
 690                        "object_type": other_obj_info["object_type"],
 691                        "distance": distance
 692                    })
 693            obj_info["distances"] = distances

 695        return objects_info

 769        """
 770        @public
 771
 772        Generate a random question vector for relational or non-relational questions.
 773
 774        As per the paper, the question vector is an 11-dimensional one-hot vector composed of:
 775        - A 6-bit one-hot vector for the color.
 776        - A 2-bit one-hot vector for the question subtype.
 777        - A 3-bit one-hot vector for the specific question.
 778
 779        Parameters:
 780        -----------
 781        relational : bool, optional
 782            Whether the question vector is for a relational question (True) or non-relational question (False).
 783
 784        Returns:
 785        --------
 786        list
 787            An 11-dimensional list representing the question vector.
 788        """
 789        # as per the paper, the question vector is a 11-dimensional one-hot vector, composed of 6-bit one-hot vector for the color, 
 790        # 2-bit one-hot vector of the question subtype and 3-bit one-hot for the specific question
 791        vector = [0] * 11
 792
 793        six_bit_index = np.random.randint(0, 6)
 794        three_bit_index = np.random.randint(0, 3) + 8
 795
 796        vector[six_bit_index] = 1
 797        if relational:
 798            vector[6] = 1
 799        else:
 800            vector[7] = 1
 801        vector[three_bit_index] = 1
 802
 803        return vector

1069    def _get_non_relational_answer(self, img_dim, subtype, color, objects_info):
1070        """
1071        @public
1072
1073        Determine the answer to a non-relational question based on its subtype and reference color.
1074
1075        Parameters:
1076        -----------
1077        img_dim : int
1078            The dimension (width and height) of the image.
1079        subtype : str
1080            The subtype of the question (e.g., "topbottom", "leftright", "shape").
1081        color : str
1082            The color of the reference object for the question.
1083        objects_info : list
1084            A list of dictionaries, where each dictionary contains information about an object 
1085            in the image, including its color, shape, and bounding box location.
1086
1087        Returns:
1088        --------
1089        str
1090            The answer to the non-relational question. This could be "top", "bottom", "left", "right",
1091            or the shape of the object, depending on the subtype of the question.
1092
1093        Raises:
1094        -------
1095        ValueError
1096            If the subtype does not correspond to any known question type.
1097        """
1098        if subtype == "topbottom":
1099            for obj_info in objects_info:
1100                # skip the question object
1101                if 'question' in obj_info:
1102                    continue
1103                if obj_info["color_name"] == color:
1104                    # consider the object to be on the top if its y-coordinate is less than half of the image height
1105                    # use the center of the bounding box to determine the y-coordinate
1106                    if self._compute_bbox_center(obj_info["bboxlocation"])[1] < img_dim // 2:
1107                        return "top"
1108                    else:
1109                        return "bottom"
1110        elif subtype == "leftright":
1111            for obj_info in objects_info:
1112                # skip the question object
1113                if 'question' in obj_info:
1114                    continue
1115                if obj_info["color_name"] == color:
1116                    # consider the object to be on the left if its x-coordinate is less than half of the image width
1117                    # use the center of the bounding box to determine the x-coordinate
1118                    if self._compute_bbox_center(obj_info["bboxlocation"])[0] < img_dim // 2:
1119                        return "left"
1120                    else:
1121                        return "right"
1122        elif subtype == "shape":
1123            for obj_info in objects_info:
1124                # skip the question object
1125                if 'question' in obj_info:
1126                    continue
1127                if obj_info["color_name"] == color:
1128                    return obj_info["object_type"]
1129        else:

 384        """

 386        existing_boxes = []

 395            img = self._gen_noise(img, noise_prob)

 630    def _do_overlap(self, box1, box2):
 631        """
 632        @public
 633
 634        Check if two bounding boxes overlap with an offset applied to all sides.
 635
 636        Parameters:
 637        -----------
 638        box1 : tuple
 639            A tuple containing two points that represent the first bounding box ((x1, y1), (x2, y2)).
 640        box2 : tuple
 641            A tuple containing two points that represent the second bounding box ((x1, y1), (x2, y2)).
 642
 643        Returns:
 644        --------
 645        bool
 646            True if the bounding boxes overlap, False otherwise.
 647        """
 648        # check if two bounding boxes overlap with an offset applied to all sides
 649        offset = 3
 650
 651        expanded_box1 = ((box1[0][0] - offset, box1[0][1] - offset), (box1[1][0] + offset, box1[1][1] + offset))
 652        expanded_box2 = ((box2[0][0] - offset, box2[0][1] - offset), (box2[1][0] + offset, box2[1][1] + offset))
